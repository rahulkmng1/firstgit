drop database ecotrack;

create database ecotrack;

use  ecotrack;

-- USER table

CREATE TABLE user (

    UserID INT PRIMARY KEY,

    Name VARCHAR(100) NOT NULL,

    Role ENUM('Admin','Auditor','Officer_Environmental','Officer_Pollution_Control','Citizen','Manager') NOT NULL,

    Email VARCHAR(100) UNIQUE NOT NULL,

    passwordHash VARCHAR(100) NOT NULL,

    Phone VARCHAR(20),

    Status VARCHAR(20) NOT NULL

);

-- RESOURCE table (linked to Officer_Pollution_Control)

CREATE TABLE resource (

    ResourceID INT PRIMARY KEY,

    OfficerPollutionControlID INT,

    Type ENUM('Human','Financial','Material','Technological'),

    Capacity DECIMAL(15,2),

    Usages DECIMAL(15,2),

    Status VARCHAR(50),

    FOREIGN KEY (OfficerPollutionControlID) REFERENCES user(UserID)

);

-- ENVIRONMENTAL DATA LOG table (linked to Officer_Environmental)

CREATE TABLE environmental_data_log (

    EntryID INT PRIMARY KEY,

    OfficerEnvironmentalID INT,

    CitizenID INT,

    Type ENUM('Air','Water','Soil','Noise','General'),

    Location VARCHAR(255),

    Value DECIMAL(10,2),

    Date DATETIME,

    Status VARCHAR(50),

    FOREIGN KEY (OfficerEnvironmentalID) REFERENCES user(UserID),

    FOREIGN KEY (CitizenID) REFERENCES user(UserID)

);

-- PROJECT table (dependent on Audit as well)

CREATE TABLE project (

    ProjectID INT PRIMARY KEY,

    ProjectManagerID INT,

    ResourceID INT,

    EntryID INT,

    ComplianceID INT,

    AuditID INT,

    Title VARCHAR(255) NOT NULL,

    Description TEXT,

    StartDate DATE,

    EndDate DATE,

    Budget DECIMAL(15,2),

    Status VARCHAR(50)

);

-- REPORT table

CREATE TABLE report (

    ReportID INT PRIMARY KEY,

    Scope ENUM('Air','Water','Soil','Noise','General') NOT NULL,

    Metrics TEXT,

    GeneratedDate DATETIME,

    Status VARCHAR(50)

);

-- MILESTONE table

CREATE TABLE milestone (

    MilestoneID INT PRIMARY KEY,

    ProjectID INT,

    ProjectManagerID INT,

    Title VARCHAR(255),

    Date DATE,

    Status VARCHAR(50),

    FOREIGN KEY (ProjectID) REFERENCES project(ProjectID),

    FOREIGN KEY (ProjectManagerID) REFERENCES user(UserID)

);

-- DOCUMENTUPLOAD table

CREATE TABLE documentupload (

    DocumentID INT PRIMARY KEY,

    OfficerEnvironmentalID INT,

    EntryID INT,

    FileURI VARCHAR(512),

    UploadedDate DATETIME,

    VerificationStatus TINYINT(1),

    FOREIGN KEY (OfficerEnvironmentalID) REFERENCES user(UserID),

    FOREIGN KEY (EntryID) REFERENCES environmental_data_log(EntryID)

);

-- POLLUTIONSOURCE table

CREATE TABLE pollutionsource (

    SourceID INT PRIMARY KEY,

    OfficerPollutionControlID INT,

    Type ENUM('Industrial','Vehicular','Agricultural','Domestic','Other'),

    Location VARCHAR(255),

    Status VARCHAR(50),

    FOREIGN KEY (OfficerPollutionControlID) REFERENCES user(UserID)

);

-- ISSUE table

CREATE TABLE issue (

    IssueID INT PRIMARY KEY,

    OfficerEnvironmentalID INT,

    CitizenID INT,

    Type ENUM('Air','Water','Soil','Noise','General'),

    Location VARCHAR(255),

    Date DATETIME,

    IssueDiscription VARCHAR(200),

    Status ENUM('Open','InProgress','Resolved','Closed'),

    FOREIGN KEY (OfficerEnvironmentalID) REFERENCES user(UserID),

    FOREIGN KEY (CitizenID) REFERENCES user(UserID)

);

-- NOTIFICATION table

CREATE TABLE notification (

    NotificationID INT PRIMARY KEY,

    UserID INT,

    EntityID INT,

    Message TEXT,

    Category ENUM('Alert','Reminder','Update','Warning'),

    CreatedDate DATETIME,

    FOREIGN KEY (UserID) REFERENCES user(UserID)

);

CREATE TABLE impact (

    ImpactID INT PRIMARY KEY,

    ProjectID INT,

    ProjectManagerID INT,

    MetricsJSON JSON,

    Date DATE,

    Status VARCHAR(50),

    FOREIGN KEY (ProjectID) REFERENCES project(ProjectID),

    FOREIGN KEY (ProjectManagerID) REFERENCES user(UserID)

);

CREATE TABLE audit (

    AuditID INT PRIMARY KEY,

    AuditorID INT,

    ProjectID INT,

    Scope VARCHAR(255),

    Findings TEXT,

    Date DATETIME,

    Status VARCHAR(50),

    FOREIGN KEY (AuditorID) REFERENCES user(UserID),

    FOREIGN KEY (ProjectID) REFERENCES project(ProjectID)

);

-- COMPLIANCERECORD table

CREATE TABLE compliancerecord (

    ComplianceID INT PRIMARY KEY,

    AuditorID INT,

    ProjectID INT,

    MilestoneID INT,

    ImpactID INT,

    Type ENUM('Safety','Environmental','Legal','Financial'),

    Result VARCHAR(100),

    Date DATETIME,

    Notes TEXT,

    FOREIGN KEY (AuditorID) REFERENCES user(UserID),

    FOREIGN KEY (ProjectID) REFERENCES project(ProjectID),

    FOREIGN KEY (MilestoneID) REFERENCES milestone(MilestoneID),

    FOREIGN KEY (ImpactID) REFERENCES impact(ImpactID)

);

-- FEEDBACK table

CREATE TABLE feedback (

    FeedbackID INT PRIMARY KEY,

    CitizenID INT,

    ProjectID INT,

    Comments TEXT,

    Date DATETIME,

    Status VARCHAR(50),

    FOREIGN KEY (CitizenID) REFERENCES user(UserID),

    FOREIGN KEY (ProjectID) REFERENCES project(ProjectID)

-- AUDITLOG table

CREATE TABLE auditlog (

    AuditLogID INT PRIMARY KEY,

    UserID INT,

    Action VARCHAR(255),

    Resource VARCHAR(100),

    Timestamp DATETIME,

    FOREIGN KEY (UserID) REFERENCES user(UserID)

);
